# Fix instability, then train

    Action(action="adjust_learning_rate")
)

    Action(action="run_training_epoch")
)

## Technical Highlights

### Realistic Simulation
```python
# Instability causes registration
if instability:
    road_iou -= 0.05 + random()*0.03

# Augmentation quantified
if augmentation:
    improvement *= 1.10  # +10%

# Sky class only learns with balancing
if class_balancing:
    sky_iou += improvement * 0.5
else:
    sky_iou = 0.0  # Stays zero

# Diminishing returns
improvement = base * (1 - progress_factor * 0.6)
```

### Multi-Component Rewards
```python
reward = (
    iou_improvement * 10.0 +
    sky_coverage * 2.0 +
    efficiency_bonus * 0.5 -
    instability_penalty * 1.0 -
    overfitting_penalty * 0.5
)
```

### Graduated Grading
```python
# Hard task (4 weighted objectives)
score = (
    min(1.0, iou / 0.65) * 0.40 +      # IoU (40%)
    min(1.0, sky / 0.50) * 0.30 +      # Sky coverage (30%)
    compute_efficiency * 0.15 +         # Efficiency (15%)
    no_instability * 0.15               # Stability (15%)
)
```
